# =============================================================================
# CEO Weekly Briefing - Cron Job Configuration
# =============================================================================
# Runs every Sunday at 8:00 AM to generate weekly briefing
# =============================================================================

# -----------------------------------------------------------------------------
# INSTALLATION
# -----------------------------------------------------------------------------
# Run this command to install the cron job:
#
#   crontab -l 2>/dev/null | cat - ceo_briefing_cron.txt | crontab -
#
# Or manually run:
#   crontab -e
# Then paste the lines below
# -----------------------------------------------------------------------------

# -----------------------------------------------------------------------------
# Environment Setup
# -----------------------------------------------------------------------------
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
MAILTO=""

# -----------------------------------------------------------------------------
# Cron Jobs
# -----------------------------------------------------------------------------

# CEO Weekly Briefing: Every Sunday at 8:00 AM
# Generates briefing for the previous week (Monday to Sunday)
0 8 * * 0 cd /path/to/Hack-03-AI-Employee-giaic && ./venv/bin/python scripts/ceo_briefing.py >> logs/ceo_briefing.log 2>&1

# Optional: Send briefing via email (if email script available)
# 30 8 * * 0 cd /path/to/Hack-03-AI-Employee-giaic && ./venv/bin/python scripts/send_briefing_email.py >> logs/ceo_briefing.log 2>&1

# Optional: Clean old briefings quarterly (keep last 12 weeks)
# 0 9 1 1,4,7,10 * find /path/to/Hack-03-AI-Employee-giaic/AI_Employee_Vault/Briefings -name "*CEO_Briefing.md" -mtime +90 -delete

# -----------------------------------------------------------------------------
# ALTERNATIVE SCHEDULES

# Every Sunday at 9:00 AM (later)
# 0 9 * * 0 cd /path/to/Hack-03-AI-Employee-giaic && ./venv/bin/python scripts/ceo_briefing.py >> logs/ceo_briefing.log 2>&1

# Every Monday at 7:00 AM (for previous week)
# 0 7 * * 1 cd /path/to/Hack-03-AI-Employee-giaic && ./venv/bin/python scripts/ceo_briefing.py >> logs/ceo_briefing.log 2>&1
